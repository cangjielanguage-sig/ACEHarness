"""Stress Test Iteration 4 - Cangjie Compiler
Focus: new compilation paths, error recovery, full pipeline AST release, expression depth
"""
import subprocess, time, os, sys, csv, json, random

CJC = "/root/ace/cangjie_compiler/build/build/bin/cjc"
ENVSETUP = "/root/ace/cangjie_compiler/output/envsetup.sh"
GEN_DIR = "/root/ace/cangjie_compiler/.ace-outputs/run-20260224145111262/stress-tests/generated_iter4"
RES_DIR = "/root/ace/cangjie_compiler/.ace-outputs/run-20260224145111262/stress-tests/results_iter4"
TIMEOUT = 180

def get_env():
    out = subprocess.check_output(["bash", "-c", f"source {ENVSETUP} && env"], text=True)
    env = {}
    for line in out.splitlines():
        if "=" in line:
            k, _, v = line.partition("=")
            env[k] = v
    return env

ENV = get_env()

def measure_compile(name, src_file, extra_flags=None, timeout=TIMEOUT):
    flags = ["--emit-chir", "-o", "/dev/null", "-Woff", "unused"]
    if extra_flags:
        flags = extra_flags + [src_file]
    else:
        flags = flags + [src_file]
    cmd = [CJC] + flags
    peak_rss = 0
    try:
        proc = subprocess.Popen(cmd, stdout=subprocess.DEVNULL, stderr=subprocess.PIPE, env=ENV)
        start = time.monotonic()
        while proc.poll() is None:
            elapsed = time.monotonic() - start
            if elapsed > timeout:
                proc.kill()
                proc.wait()
                return {"name": name, "status": "TIMEOUT", "time_ms": int(elapsed*1000),
                        "peak_rss_mb": peak_rss // 1024, "exit_code": -9}
            try:
                with open(f"/proc/{proc.pid}/status") as f:
                    for line in f:
                        if line.startswith("VmHWM:"):
                            hwm = int(line.split()[1])
                            if hwm > peak_rss: peak_rss = hwm
                            break
            except (FileNotFoundError, ProcessLookupError):
                pass
            time.sleep(0.02)
        elapsed = time.monotonic() - start
        rc = proc.returncode
        stderr_out = proc.stderr.read().decode(errors='replace')[:500]
        if rc == -11 or rc == 139: status = "SIGSEGV"
        elif rc != 0: status = "COMPILE_ERR"
        else: status = "PASS"
        return {"name": name, "status": status, "time_ms": int(elapsed*1000),
                "peak_rss_mb": peak_rss // 1024, "exit_code": rc, "stderr": stderr_out}
    except Exception as e:
        return {"name": name, "status": "ERROR", "time_ms": 0, "peak_rss_mb": 0,
                "exit_code": -1, "stderr": str(e)}

def write_src(name, content):
    path = os.path.join(GEN_DIR, f"{name}.cj")
    with open(path, "w") as f: f.write(content)
    return path

results = []

def run_test(name, src, extra_flags=None, timeout=TIMEOUT):
    path = write_src(name, src)
    r = measure_compile(name, path, extra_flags=extra_flags, timeout=timeout)
    results.append(r)
    icon = {"PASS":"OK","SIGSEGV":"CRASH","COMPILE_ERR":"ERR","TIMEOUT":"TOUT"}.get(r["status"],"???")
    print(f"  [{icon:5s}] {name:50s} {r['time_ms']:7d}ms  {r['peak_rss_mb']:5d}MB  rc={r.get('exit_code','?')}")
    return r

ENV = {}

    if "=" in line:
        k, _, v = line.partition("=")
        ENV[k] = v

def measure(name, src_file, extra_flags, timeout=300):
    cmd = [CJC] + extra_flags + [src_file]
    peak_rss = 0
    proc = subprocess.Popen(cmd, stdout=subprocess.DEVNULL, stderr=subprocess.PIPE, env=ENV)
    start = time.monotonic()
    while proc.poll() is None:
        elapsed = time.monotonic() - start
        if elapsed > timeout:
            proc.kill(); proc.wait()
            return name, "TIMEOUT", int(elapsed*1000), peak_rss // 1024
        try:
