    51→void IncrementalCompilerInstance::SaveCompilationResult(Cangjie::IncreResult&& result)
    52→{
    53→    kind = result.kind;
    54→    rawMangleName2DeclMap = result.mangle2decl;
    55→    switch (kind) {
    56→        case IncreKind::ROLLBACK:
    57→            declsToBeReCompiled.clear();
    58→            [[fallthrough]];
    59→        case IncreKind::INCR:
    60→            cachedInfo = std::move(result.cacheInfo);
    61→            CacheCompileArgs();
    62→            declsToBeReCompiled = std::move(result.declsToRecompile);
    63→            rawIncrRemovedDecls = std::move(result.deleted);
    64→            // Copy a non-const version for 'Sema' usage.
    65→            for (auto& [mangle, decl] : result.mangle2decl) {
    66→                if (!decl->TestAttr(Attribute::IMPORTED)) {
    67→                    mangledName2DeclMap.emplace(mangle, Ptr(const_cast<Decl*>(decl.get())));
    68→                }
    69→            }
    70→            for (auto changed : declsToBeReCompiled) {
    71→                if (changed->TestAttr(Attribute::IMPORTED) && (changed->IsFunc() || IsGlobalOrStaticVar(*changed))) {
    72→                    cacheMangles.EmplaceImportedInlineDeclPtr(*changed);
    73→                } else {
    74→                    Sema::CollectRemovedManglesForReCompile(*changed, cachedInfo.semaInfo,
    75→                        cacheMangles.incrRemovedDecls);
    76→                }
    77→            }
    78→            for (auto removed : rawIncrRemovedDecls) {
    79→                Sema::CollectRemovedMangles(removed, cachedInfo.semaInfo, cacheMangles.incrRemovedDecls);
    80→            }
    81→            // NOTE: since 'declsToBeReCompiled' and 'rawIncrRemovedDecls' have been process,
    82→            // this loop will only remove the re-boxed built-in types' related decls.
    83→            for (auto typeMangle : result.reBoxedTypes) {
    84→                Sema::CollectRemovedMangles(typeMangle, cachedInfo.semaInfo, cacheMangles.incrRemovedDecls);
    85→            }
    86→            break;
    87→        case IncreKind::NO_CHANGE:
    88→            cachedInfo.bitcodeFilesName = std::move(result.cacheInfo.bitcodeFilesName);
    89→            break;
    90→        case IncreKind::EMPTY_PKG:
    91→            break;
    92→        case IncreKind::INVALID:
    93→            CJC_ABORT();
    94→            break;
    95→    }
    96→}

   100→std::optional<size_t> GetCachedSplitNum(const std::string& cachedCodeGenPath)
   101→{
   102→    std::optional<size_t> splitNum{std::nullopt};
   103→    llvm::SMDiagnostic err;
   104→    auto context = llvm::LLVMContext();
   105→#ifdef _WIN32
   106→    auto tempPath = StringConvertor::NormalizeStringToUTF8(cachedCodeGenPath);
   107→    CJC_ASSERT(tempPath.has_value() && "Incorrect file name encoding.");
   108→    auto module = llvm::parseIRFile(tempPath.value(), err, context);
   109→#else
   110→    auto module = llvm::parseIRFile(cachedCodeGenPath, err, context);
   111→#endif
   112→    if (module == nullptr) {
   113→        Errorln("Illegal cached codegen info detected during incremental compilation.");
   114→        return splitNum;
   115→    }
   116→    auto splitNumMD = module->getNamedMetadata("splitNum");
   117→    for (llvm::MDNode* mdNode : splitNumMD->operands()) {
   118→        auto numMD = llvm::cast<llvm::ConstantAsMetadata>(mdNode->getOperand(0));
   119→        splitNum = llvm::cast<llvm::ConstantInt>(numMD->getValue())->getZExtValue();
   120→    }
   121→    return splitNum;
   122→}

   124→bool CheckCachedBitcode(const std::string& packageName, size_t splitNum, const GlobalOptions& globalOp)
   125→{
   126→    if (splitNum == 0) {
   127→        return false;
   128→    }
   129→    for (size_t i = 0; i < splitNum; ++i) {
   130→        std::string cachedBcPath =
   131→            globalOp.GenerateCachedPathNameForCodeGen(std::to_string(i) + '-' + packageName, "_cache.bc");
   132→        if (!FileUtil::FileExist(cachedBcPath)) {
   133→            IncrementalCompilationLogger::GetInstance().LogLn(
   134→                "the cached bitcode file is lost." + cachedBcPath + " index " + std::to_string(i));
   135→            return false;
   136→        }
   137→    }
   138→    return true;
   139→}

   142→{

   157→            "load cached bitcode file failed, roll back to full compilation");
   158→        return false;
   159→    }

   192→bool IncrementalCompilerInstance::PerformIncrementalScopeAnalysis()
   193→{
   194→#ifdef CANGJIE_CODEGEN_CJNATIVE_BACKEND
   195→    Utils::ProfileRecorder recorder("Main Stage", "AST Diff");
   196→    CJC_ASSERT(srcPkgs.size() == 1);
   197→    auto& package = srcPkgs[0];
   198→
   199→    // calculate the AST cache info
   200→    bool res = CalculateASTCache();
   201→    if (!res) {
   202→        return true;
   203→    }
   204→
   205→    auto& options = invocation.globalOptions;
   206→#ifdef CANGJIE_CODEGEN_CJNATIVE_BACKEND
   207→    if (!options.pluginPaths.empty()) {
   208→        kind = IncreKind::ROLLBACK;
   209→        Warningf("`--incremental-compile` is incompatible with `--plugin` and will be ignored.\n");
   210→        return true;
   211→    }
   212→#endif
   213→
   214→    if (!declsWithDuplicatedRawMangleName.empty()) {
   215→        kind = IncreKind::ROLLBACK;
   216→        IncrementalCompilationLogger::GetInstance().LogLn(
   217→            "illegal source code leads to duplicated raw mangle name, roll back to full compilation");
   218→        return true;
   219→    }
   220→
   221→    auto increRes = IncrementalScopeAnalysis({rawMangleName2DeclMap, std::move(astCacheInfo), *package, options,
   222→                                                 importManager, cachedInfo, fileMap, std::move(directExtends)});
   223→    if (increRes.kind == IncreKind::INCR) {
   224→        // check cached bitcode infos
   225→        const std::string& packageName = package->fullPackageName;
   226→        bool isCachedBitcodeGood = CheckCachedBitcodeInfo(packageName, options);
   227→        if (!isCachedBitcodeGood) {
   228→            increRes.kind = IncreKind::ROLLBACK;
   229→        } else {
   230→            // set deleted mangles Name for codegen
   231→            cacheMangles.incrRemovedDecls.insert(
   232→                increRes.deletedMangleNames.begin(), increRes.deletedMangleNames.end());
   233→        }
   234→    }
   235→    if (increRes.kind == IncreKind::NO_CHANGE) {
   236→        // check cached obj file
   237→        for (auto& srcPkg : GetSourcePackages()) {
   238→            options.UpdateCachedDirName(srcPkg->fullPackageName);
   239→        }
   240→        if (!options.IsLTOEnabled() && !CheckCachedObjFile(options, increRes.cacheInfo)) {
   241→            increRes.kind = IncreKind::ROLLBACK;
   242→        }
   243→        if (options.IsLTOEnabled() && !CheckCachedBitcodeInfoForLTO(options, increRes.cacheInfo)) {
   244→            increRes.kind = IncreKind::ROLLBACK;
   245→        }
   246→    }
   247→
   248→    increRes.Dump();
   249→    SaveCompilationResult(std::move(increRes));
   250→#endif
   251→    return true;
   252→}

   254→static bool HasSimpleInitialiser(const AST::VarDecl& decl)
   255→{
   256→    auto& init = decl.initializer;
   257→    if (!init) {
   258→        return false;
   259→    }
   260→    if (init->astKind == ASTKind::LIT_CONST_EXPR) {
   261→        auto kind = StaticCast<LitConstExpr>(init.get())->kind;
   262→        return kind != LitConstKind::JSTRING && kind != LitConstKind::UNIT;
   263→    }
   264→    return false;
   265→}

   267→bool IncrementalCompilerInstance::PerformSema()
   268→{
   269→    Utils::ProfileRecorder recorder("Main Stage", "Semantic");
   270→    if (auto& logger = IncrementalCompilationLogger::GetInstance(); logger.IsEnable()) {
   271→        std::string message = kind == IncreKind::NO_CHANGE
   272→            ? "no change, skip sema"
   273→            : (kind == IncreKind::ROLLBACK ? "full sema" : "incremental sema");
   274→        logger.LogLn(message);
   275→    }
   276→    if (kind == IncreKind::NO_CHANGE || kind == IncreKind::EMPTY_PKG) {
   277→        return true;
   278→    } else if (kind == IncreKind::ROLLBACK) {
   279→        return CompilerInstance::PerformSema();
   280→    }
   281→    CJC_NULLPTR_CHECK(typeManager);
   282→    // Load cached types. Should only exist exactly one source package.
   283→    for (auto srcPkg : GetSourcePackages()) {
   284→        srcPkg->EnableAttr(Attribute::INCRE_COMPILE);
   285→        Sema::HandleCtorForIncr(*srcPkg, mangledName2DeclMap, cachedInfo.semaInfo);
   286→        cacheMangles.incrRemovedDecls.merge(importManager.LoadCachedTypeForPackage(*srcPkg, mangledName2DeclMap));
   287→    }
   288→    Sema::MarkIncrementalCheckForCtor(declsToBeReCompiled);
   289→
   290→    auto visit = [](Decl& decl) {
   291→        // NOTE: Avoid some sema rules to be rechecked.
   292→        // To guarantee the member decls will be checked, the type decl cannot be marked with following attributes.
   293→        if (!decl.IsNominalDecl() && decl.astKind != ASTKind::PROP_DECL) {
   294→            decl.EnableAttr(Attribute::IS_CHECK_VISITED, Attribute::INITIALIZED, Attribute::INITIALIZATION_CHECKED,
   295→                Attribute::INCRE_COMPILE);
   296→        }
   297→        // Remove body to avoid walker walk-in.
   298→        if (auto fd = DynamicCast<FuncDecl*>(&decl); fd && fd->funcBody) {
   299→            fd->funcBody->body.reset();
   300→            return;
   301→        }
   302→        if (auto vd = DynamicCast<VarDecl*>(&decl)) {
   303→            if (!vd->initializer) {
   304→                return;
   305→            }
   306→            vd->EnableAttr(Attribute::DEFAULT);
   307→            if (HasSimpleInitialiser(*vd)) {
   308→                vd->DisableAttr(Attribute::IS_CHECK_VISITED);
   309→            }
   310→        }
   311→        // PropDecl's getter/setter will be counted separately and should not be cleared from propDecl.
   312→    };
   313→    for (auto& [_, decl] : mangledName2DeclMap) {
   314→        if (!decl->toBeCompiled && !IsInDeclWithAttribute(*decl, Attribute::GENERIC) && Ty::IsTyCorrect(decl->ty) &&
   315→            (decl->astKind != ASTKind::FUNC_DECL || !decl->TestAttr(Attribute::DEFAULT))) {
   316→        // NOTE: for now, generic definitions and default implementations should be kept for code re-generation.
   317→            visit(*decl);
   318→        }
   319→    }
   320→    return CompilerInstance::PerformSema();
   321→}

   351→            IterateAllExportableDecls(*srcPkg, [this](auto& decl) {
   352→                // If non-export function/variable is marked as external linkage, codegen should change its linkage.
   353→                if (!decl.IsExportedDecl() && decl.linkage == Linkage::EXTERNAL && !decl.mangledName.empty()) {
   354→                    cacheMangles.newExternalDecls.emplace(decl.mangledName);
   355→                }
   356→            });
   357→        }
   358→    }

   375→void IncrementalCompilerInstance::UpdateCachedInfo()
   376→{
   377→    UpdateMangleNameForCachedInfo();
   378→    UpdateCHIROptEffectMap();
   379→    if (kind == IncreKind::ROLLBACK) {
   380→        cachedInfo.virtualFuncDep = std::move(chirInfo.curVirtFuncWrapDep);
   381→        cachedInfo.ccOutFuncs = std::move(chirInfo.ccOutFuncsRawMangle);
   382→        cachedInfo.varInitDepMap = std::move(chirInfo.varInitDepMap);
   383→    } else if (kind == IncreKind::INCR) {
   384→        for (auto& [raw, _] : chirInfo.delVirtFuncWrapForIncr) {
   385→            cachedInfo.virtualFuncDep.erase(raw);
   386→        }
   387→        for (auto& [raw, warp] : chirInfo.curVirtFuncWrapDep) {
   388→            cachedInfo.virtualFuncDep[raw] = warp;
   389→        }
   390→        for (auto& rm : std::as_const(rawIncrRemovedDecls)) {
   391→            cachedInfo.varInitDepMap.erase(rm);
   392→        }
   393→        for (auto& [raw, warp] : chirInfo.varInitDepMap) {
   394→            cachedInfo.varInitDepMap[raw] = warp;
   395→        }
   396→        cachedInfo.ccOutFuncs.merge(std::move(chirInfo.ccOutFuncsRawMangle));
   397→    }
   398→    // Update usage of compiler added.
   399→    for (auto pkg : GetSourcePackages()) {
   400→        Sema::CollectCompilerAddedDeclUsage(*pkg, cachedInfo.semaInfo);
   401→    }
   402→}

   468→bool IncrementalCompilerInstance::PerformASTRelease()
   469→{
   470→    Utils::ProfileRecorder recorder("Main Stage", "Release AST");
   471→
   472→    // Always extract CodeGen metadata before potential release.
   473→    PrepareCodeGenInfo();
   474→
   475→    // For NO_CHANGE / EMPTY_PKG, no AST was built — nothing to release.
   476→    if (kind == IncreKind::NO_CHANGE || kind == IncreKind::EMPTY_PKG) {
   477→        return true;
   478→    }
   479→
   480→    // Save results before AST is freed (ADR-009, F-02).
   481→    // PerformResultsSaving needs srcPkgs for ExportASTSignature and SaveCjo.
   482→    if (!resultsSaved) {
   483→        if (!PerformResultsSaving()) {
   484→            return false;  // Propagate save failure to abort compilation.
   485→        }
   486→        resultsSaved = true;
   487→    }
   488→
   489→    // Write incremental compilation cache before AST is freed (ADR-009).
   490→    if (ShouldWriteCacheFile() && !cacheWritten) {
   491→        UpdateCachedInfo();
   492→        WriteCachedInfo();
   493→        cacheWritten = true;
   494→    }
   495→
   496→    // Allow skipping AST release for debugging (e.g., ASan leak analysis).
   497→    if (invocation.globalOptions.noReleaseAST) {
   498→        return true;
   499→    }
   500→
   501→    // Clean incremental-compilation-specific AST pointers.
   502→    changedTypes.clear();
   503→    declsToBeReCompiled.clear();
   504→    cacheMangles.Clear();
   505→
   506→    // Delegate to base class for the 5-phase cleanup.
   507→    ReleaseASTMemory();
   508→    return true;
   509→}

   520→bool IncrementalCompilerInstance::PerformCodeGen()
   521→{
   522→    Utils::ProfileRecorder recorder("Main Stage", "CodeGen");
   523→    if (kind == IncreKind::NO_CHANGE) {
   524→        // Use codeGenPkgInfos if AST is released, otherwise use srcPkgs.
   525→        if (IsASTReleased()) {
   526→            for (auto& info : codeGenPkgInfos) {
   527→                LoadCachedCodegenResult();
   528→                invocation.globalOptions.UpdateCachedDirName(info.fullPackageName);
   529→            }
   530→        } else {
   531→            for (auto& srcPkg : GetSourcePackages()) {
   532→                LoadCachedCodegenResult();
   533→                invocation.globalOptions.UpdateCachedDirName(srcPkg->fullPackageName);
   534→            }
   535→        }

   571→{
   572→    // It can be not null in case of part of the package was deserialized from .chir
   573→    bool needDesCHIR = opts.inputChirFiles.size() != 0;
   574→    if (!needDesCHIR) {
   575→        package = builder.CreatePackage(node.fullPackageName);
   576→        outputCHIR = opts.outputMode == GlobalOptions::OutputMode::CHIR;
   577→    } else if (TryToDeserializeCHIR()) {
   578→        package = builder.GetCurPackage();
   579→        BuildDeserializedTable();
   580→    } else {
   581→        return false;
   582→    }
   583→    // Translating common alongside with merging specific is not supported
   584→    CJC_ASSERT(!(outputCHIR && mergingSpecific));
   585→    package->SetPackageAccessLevel(BuildPackageAccessLevel(node.accessible));
   586→    RegisterAllSources();
   587→    CJC_NULLPTR_CHECK(package);
   588→    dependencyPkg = importManager.GetAllDependentPackageNames(node.fullPackageName);
   589→
   590→    // step 1: collect all top-level decls
   591→    CollectTopLevelDecls(node);
   592→
   593→    // step 2: sort and do circular dependency checking
   594→    auto [initOrder, result] = SortGlobalVarDecl(node);
   595→    if (!result) {
   596→        return false;
   597→    }

   600→    CacheSomeDeclsToGlobalSymbolTable();

   602→    // step 4: translate all colleted decls
   603→    TranslateAllDecls(node, initOrder);
   604→
   605→    // step 5: ast2chir check
   606→    AST2CHIRCheck();
