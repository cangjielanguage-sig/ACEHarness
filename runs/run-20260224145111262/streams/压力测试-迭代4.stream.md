"""Stress Test Iteration 3 - Cangjie Compiler
Focus: closure cliff pinpoint, larger scale, fuzz testing, previous findings re-verify

ENVSETUP = "/root/ace/cangjie_compiler/output/envsetup.sh"

RES_DIR = "/root/ace/cangjie_compiler/.ace-outputs/run-20260224145111262/stress-tests/results_iter3"
TIMEOUT = 180

def get_env():
    out = subprocess.check_output(["bash", "-c", f"source {ENVSETUP} && env"], text=True)
    env = {}
    for line in out.splitlines():
        if "=" in line:
            k, _, v = line.partition("=")
            env[k] = v
    return env

ENV = get_env()

def measure_compile(name, src_file, extra_flags=None, timeout=TIMEOUT):
    flags = ["--emit-chir", "-o", "/dev/null", "-Woff", "unused"]
    if extra_flags:
        flags.extend(extra_flags)
    cmd = [CJC] + flags + [src_file]
    peak_rss = 0
    try:
        proc = subprocess.Popen(cmd, stdout=subprocess.DEVNULL, stderr=subprocess.PIPE, env=ENV)
        start = time.monotonic()
        while proc.poll() is None:
            elapsed = time.monotonic() - start
            if elapsed > timeout:
                proc.kill()
                proc.wait()
                return {"name": name, "status": "TIMEOUT", "time_ms": int(elapsed*1000),
                        "peak_rss_mb": peak_rss // 1024, "exit_code": -9}
            try:
                with open(f"/proc/{proc.pid}/status") as f:
                    for line in f:
                        if line.startswith("VmHWM:"):
                            hwm = int(line.split()[1])
                            if hwm > peak_rss:
                                peak_rss = hwm
                            break
            except (FileNotFoundError, ProcessLookupError):
                pass
            time.sleep(0.02)
        elapsed = time.monotonic() - start
        rc = proc.returncode
        # Final HWM read
        stderr_out = proc.stderr.read().decode(errors='replace')[:500]
        if rc == -11 or rc == 139:
            status = "SIGSEGV"
        elif rc != 0:
            status = "COMPILE_ERR"
        else:
            status = "PASS"
        return {"name": name, "status": status, "time_ms": int(elapsed*1000),
                "peak_rss_mb": peak_rss // 1024, "exit_code": rc, "stderr": stderr_out}
    except Exception as e:
        return {"name": name, "status": "ERROR", "time_ms": 0, "peak_rss_mb": 0,
                "exit_code": -1, "stderr": str(e)}

def write_src(name, content):
    path = os.path.join(GEN_DIR, f"{name}.cj")
    with open(path, "w") as f:
        f.write(content)
    return path

def gen_closures(n):
    """Generate n closures with captures"""
    lines = ["package test"]
    lines.append("func wrapper(): Int64 {")
    lines.append("  var total: Int64 = 0")
    for i in range(n):
        lines.append(f"  let c{i} = {{ => total + {i} }}")
    # Sum all closures
    parts = [f"c{i}()" for i in range(min(n, 50))]  # limit sum to avoid huge expr
    lines.append(f"  total = {' + '.join(parts)}")
    lines.append("  return total")
    lines.append("}")
    lines.append("main(): Int64 { return wrapper() }")
    return "\n".join(lines)
